## Install

```bash
npm i detect-facebook-in-app-browser # npm

yarn add detect-facebook-in-app-browser # yarn
```

## Usage

```ts
import isFacebookInAppBrowser from 'detect-facebook-in-app-browser';

if (isFacebookInAppBrowser()) {
  // do something...
}
```
